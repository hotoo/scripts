/**
 * 迭代器。
 * @param {Array} array 将指定数组转换为迭代器。
 * @version 0.1, 2007/11/15
 */
var Iterator = function(array){
	var _arr = array;
	var _index = 0;
	
	/**
	 * 判断迭代器是否还有下一个元素。
	 * @return {Boolean} true, 如果有，否则，false。
	 */
	this.hasNext = function(){
		return _index < _arr.length;
	};
	
	/**
	 * 获取迭代器中下一个元素。
	 * @return {Object}
	 */
	this.next = function(){
		return _arr[_index++];
	};
};
