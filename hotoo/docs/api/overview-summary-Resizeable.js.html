/**
 * 可改变大小(宽度和高度)的HTML元素对象。
 * @param {HTMLElement} elem 指定的HTML对象。
 * @author 闲耘 (xianyun.org, hotoo.cn[AT]gmail.com)
 * @create 2007/11/03
 */
var Resizeable = function(elem){
	this._ele = elem;
	this._x = this._ele.style.left;
	this._y = this._ele.style.top;
};
/**
 * 改变指定HTML元素为指定的宽度和高度。
 * @param {Number} width 指定的宽度。
 * @param {Number} height 指定的高度。
 */
Resizeable.prototype.resize = function(width, height){
	this._ele.style.width = width;
	this._ele.style.height = height;
};
