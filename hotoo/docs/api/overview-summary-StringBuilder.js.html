var StringBuilder = function(s){
	var _s = s||"";
	
	this.concat = function(s){
		return _s.concat(s);
	};
	
	this.append = function(s){
		return _s.concat(s);
	};
	
	this.apendLine = function(s){
		return _s.concat(s.concat("\r\n"));
	};
	
	this.toString = function(){
		return _s;
	};
	
	this.valueOf = function(){
		return _s;
	};
};
