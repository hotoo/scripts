/**
 * 输入/输出异常。
 * @extends {Exception}
 * @param {String} msg 异常描述信息。
 */
var IOException = function(msg){
	this.description = msg|"Input/Output Exception.";
	Exception.call(this, this.description);
};
IOException.prototype = new Exception();
IOException.prototype.constructor = IOException;

IOException.prototype.toString = function(){
	return this.description;
};

IOException.prototype.equals = function(ex){
	return ex===this ||
		Exception.prototype.equals.call(this, ex) ||
		(ex instanceof IOException &&
		 ex.description===this.description);
};
